15use partition::expr::PartitionExpr;
16use serde::{Deserialize, Serialize};
17use store_api::storage::RegionId;
18
19use crate::procedure::repartition::group::GroupId;
20
21/// Metadata describing a region involved in the plan.
22#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
23pub struct RegionDescriptor {
24    /// The region id of the region involved in the plan.
25    pub region_id: RegionId,
26    /// The partition expression of the region.
27    pub partition_expr: PartitionExpr,
28}
29
30/// A plan entry for the region allocation phase, describing source regions
31/// and target partition expressions before allocation.
32#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
33pub struct AllocationPlanEntry {
34    /// The group id for this plan entry.
35    pub group_id: GroupId,
36    /// Source region descriptors involved in the plan.
37    pub source_regions: Vec<RegionDescriptor>,
38    /// The target partition expressions for the new or changed regions.
39    pub target_partition_exprs: Vec<PartitionExpr>,
40    /// The number of regions that need to be allocated (target count - source count, if positive).
41    pub regions_to_allocate: usize,
42    /// The number of regions that need to be deallocated (source count - target count, if positive).
43    pub regions_to_deallocate: usize,
44    /// For each `source_regions[k]`, the corresponding vector contains global
45    /// `target_partition_exprs` that overlap with it.
46    pub transition_map: Vec<Vec<usize>>,
47}
48
49/// A plan entry for the dispatch phase after region allocation,
50/// with concrete source and target region descriptors.
51#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
52pub struct RepartitionPlanEntry {
53    /// The group id for this plan entry.
54    pub group_id: GroupId,
55    /// The source region descriptors involved in the plan.
56    pub source_regions: Vec<RegionDescriptor>,
57    /// The target region descriptors involved in the plan.
58    pub target_regions: Vec<RegionDescriptor>,
59    /// The region ids of the allocated regions.
60    pub allocated_region_ids: Vec<RegionId>,
61    /// The region ids of the regions that are pending deallocation.
62    pub pending_deallocate_region_ids: Vec<RegionId>,
63    /// For each `source_regions[k]`, the corresponding vector contains global
64    /// `target_regions` that overlap with it.
65    pub transition_map: Vec<Vec<usize>>,
66}
67
68impl RepartitionPlanEntry {
69    /// Converts an allocation plan entry into a repartition plan entry.
70    ///
71    /// The target regions are derived from the source regions and the target partition expressions.
72    /// The allocated region ids and pending deallocate region ids are empty.
73    pub fn from_allocation_plan_entry(
74        AllocationPlanEntry {
75            group_id,
76            source_regions,
77            target_partition_exprs,
78            regions_to_allocate,
79            regions_to_deallocate,
80            transition_map,
81        }: &AllocationPlanEntry,
82    ) -> Self {
83        debug_assert!(*regions_to_allocate == 0 && *regions_to_deallocate == 0);
84        let target_regions = source_regions
85            .iter()
86            .zip(target_partition_exprs.iter())
87            .map(|(source_region, target_partition_expr)| RegionDescriptor {
88                region_id: source_region.region_id,
89                partition_expr: target_partition_expr.clone(),
90            })
91            .collect::<Vec<_>>();
92
93        Self {
94            group_id: *group_id,
95            source_regions: source_regions.clone(),
96            target_regions,
97            allocated_region_ids: vec![],
98            pending_deallocate_region_ids: vec![],
99            transition_map: transition_map.clone(),
100        }
101    }
102}
