28/// Topic pool for kafka remote wal.
29/// Responsible for:
30/// 1. Persists topics in kvbackend.
31/// 2. Creates topics in kafka.
32/// 3. Selects topics for regions.
33pub struct KafkaTopicPool {
34    pub(crate) topics: Vec<String>,
35    // Manages topics in kvbackend.
36    topic_manager: KafkaTopicManager,
37    // Creates topics in kafka.
38    topic_creator: KafkaTopicCreator,
39    pub(crate) selector: TopicSelectorRef,
40    auto_create_topics: bool,
41}
42
43impl fmt::Debug for KafkaTopicPool {
44    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
45        f.debug_struct("KafkaTopicPool")
46            .field("topics", &self.topics)
47            .field("auto_create_topics", &self.auto_create_topics)
48            .finish()
49    }
50}
51
52impl KafkaTopicPool {
53    pub fn new(
54        config: &MetasrvKafkaConfig,
55        kvbackend: KvBackendRef,
56        topic_creator: KafkaTopicCreator,
57    ) -> Self {
58        let num_topics = config.kafka_topic.num_topics;
59        let prefix = config.kafka_topic.topic_name_prefix.clone();
60        let topics = (0..num_topics)
61            .map(|i| format!("{}_{}", prefix, i))
62            .collect();
63
64        let selector = match config.kafka_topic.selector_type {
65            TopicSelectorType::RoundRobin => RoundRobinTopicSelector::with_shuffle(),
66        };
67
68        let topic_manager = KafkaTopicManager::new(kvbackend);
69
70        Self {
71            topics,
72            topic_manager,
73            topic_creator,
74            selector: Arc::new(selector),
75            auto_create_topics: config.auto_create_topics,
76        }
77    }
78
79    /// Tries to activate the topic manager when metasrv becomes the leader.
80    /// First tries to restore persisted topics from the kv backend.
81    /// If not enough topics retrieved, it will try to contact the Kafka cluster and request creating more topics.
82    pub async fn activate(&self) -> Result<()> {
83        if !self.auto_create_topics {
84            return Ok(());
85        }
86
87        let num_topics = self.topics.len();
88        ensure!(num_topics > 0, InvalidNumTopicsSnafu { num_topics });
89
90        let topics_to_be_created = self
91            .topic_manager
92            .get_topics_to_create(&self.topics)
93            .await?;
94
95        if !topics_to_be_created.is_empty() {
96            self.topic_creator
97                .prepare_topics(&topics_to_be_created)
98                .await?;
99            self.topic_manager.persist_topics(&self.topics).await?;
100        }
101        Ok(())
102    }
103
104    /// Selects one topic from the topic pool through the topic selector.
105    pub fn select(&self) -> Result<&String> {
106        self.selector.select(&self.topics)
107    }
108
109    /// Selects a batch of topics from the topic pool through the topic selector.
110    pub fn select_batch(&self, num_topics: usize) -> Result<Vec<&String>> {
111        (0..num_topics)
112            .map(|_| self.selector.select(&self.topics))
113            .collect()
114    }
115}

117#[cfg(test)]
118mod tests {
119    use common_wal::config::kafka::common::{KafkaConnectionConfig, KafkaTopicConfig};
120    use common_wal::test_util::run_test_with_kafka_wal;
121
122    use super::*;
123    use crate::kv_backend::memory::MemoryKvBackend;
124    use crate::wal_options_allocator::topic_creator::build_kafka_topic_creator;
125
126    /// Tests that the topic manager could allocate topics correctly.
127    #[tokio::test]
128    async fn test_alloc_topics() {
129        run_test_with_kafka_wal(|broker_endpoints| {
130            Box::pin(async {
131                // Constructs topics that should be created.
132                let topics = (0..256)
133                    .map(|i| format!("test_alloc_topics_{}_{}", i, uuid::Uuid::new_v4()))
134                    .collect::<Vec<_>>();
135
136                // Creates a topic manager.
137                let kafka_topic = KafkaTopicConfig {
138                    replication_factor: broker_endpoints.len() as i16,
139                    ..Default::default()
140                };
141                let config = MetasrvKafkaConfig {
142                    connection: KafkaConnectionConfig {
143                        broker_endpoints,
144                        ..Default::default()
145                    },
146                    kafka_topic,
147                    ..Default::default()
148                };
149                let kv_backend = Arc::new(MemoryKvBackend::new()) as KvBackendRef;
150                let topic_creator = build_kafka_topic_creator(&config).await.unwrap();
151                let mut topic_pool = KafkaTopicPool::new(&config, kv_backend, topic_creator);
152                // Replaces the default topic pool with the constructed topics.
153                topic_pool.topics.clone_from(&topics);
154                // Replaces the default selector with a round-robin selector without shuffled.
155                topic_pool.selector = Arc::new(RoundRobinTopicSelector::default());
156                topic_pool.activate().await.unwrap();
157
158                // Selects exactly the number of `num_topics` topics one by one.
159                let got = (0..topics.len())
160                    .map(|_| topic_pool.select().unwrap())
161                    .cloned()
162                    .collect::<Vec<_>>();
163                assert_eq!(got, topics);
164
165                // Selects exactly the number of `num_topics` topics in a batching manner.
166                let got = topic_pool
167                    .select_batch(topics.len())
168                    .unwrap()
169                    .into_iter()
170                    .map(ToString::to_string)
171                    .collect::<Vec<_>>();
172                assert_eq!(got, topics);
173
174                // Selects more than the number of `num_topics` topics.
175                let got = topic_pool
176                    .select_batch(2 * topics.len())
177                    .unwrap()
178                    .into_iter()
179                    .map(ToString::to_string)
180                    .collect::<Vec<_>>();
181                let expected = vec![topics.clone(); 2]
182                    .into_iter()
183                    .flatten()
184                    .collect::<Vec<_>>();
185                assert_eq!(got, expected);
186            })
187        })
188        .await;
189    }
